实验任务一

GetParameterTest.java

import java.io.*;  
import javax.servlet.ServletConfig;  
import javax.servlet.annotation.WebServlet;  
import javax.servlet.http.HttpServlet;  
import javax.servlet.http.HttpServletRequest;  
import javax.servlet.http.HttpServletResponse;  

@WebServlet(name = "GetParameterTest", value = "/GetParameterTest")  
public class GetParameterTest extends HttpServlet {  
    @Override  
    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws IOException {  
        response.setContentType("text/html;charset=utf-8");  
        PrintWriter out = response.getWriter();  
        ServletConfig config = this.getServletConfig();  
        String author = config.getInitParameter("author");  
        String school = config.getInitParameter("school");  
        String name = config.getInitParameter("Servlet名");  
        out.println("school："+school+"<br>");  
        out.println("author："+author+"<br>");  
        out.println("Servlet名："+name+"<br>");  
    }  
    @Override  
    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws IOException { 
    }
}

GetWebParameter.java

import java.io.*;
import javax.servlet.ServletContext;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

@WebServlet(name = "GetWebParameter", value = "/GetWebParameter")
public class GetWebParameter extends HttpServlet {

    @Override
    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws IOException {
        response.setContentType("text/html;charset=utf-8");
        PrintWriter out = response.getWriter();
        ServletContext context = this.getServletContext();
        out.println("Web项目的所有参数如下"+"<br>");
        out.println("first：" + context.getInitParameter("first") + "<br>");
        out.println("second：" + context.getInitParameter("second"));
    }

    @Override
    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws IOException {

    }
}

GetResourceServlet

import java.io.*;
import java.util.Properties;
import javax.servlet.ServletContext;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

@WebServlet(name = "GetResourceServlet", value = "/GetResourceServlet")
public class GetResourceServlet extends HttpServlet {

    @Override
    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws IOException {
        response.setContentType("text/html;charset=utf-8");
        PrintWriter out = response.getWriter();
        ServletContext context=getServletContext();
        String path=context.getRealPath("data.properties");
        FileInputStream fis=new FileInputStream(path);
        Properties pros=new Properties();
        pros.load(fis);
        String address=pros.getProperty("address");
        String school=pros.getProperty("school");
        out.println("地址："+address+"<br>");
        out.println("学校："+school);
    }

    @Override
    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws IOException {

    }
}
